//Error less code for one LED

#include <WiFi.h>
#include <HTTPClient.h>

// WiFi credentials


const char* ssid = "shivu shinde 2";
const char* password = "123456789";

// Server URL
String serverURL = "http://192.168.1.6:3000/latest";

// Use ONLY one pin now
#define LIGHT_PIN 5   // LED connected here

void setup() {
  Serial.begin(115200);

  pinMode(LIGHT_PIN, OUTPUT);

  // LED OFF initially
  digitalWrite(LIGHT_PIN, LOW);

  Serial.println("Connecting to WiFi...");

  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nConnected!");
  Serial.print("ESP32 IP: ");
  Serial.println(WiFi.localIP());
}

void loop() {

  if (WiFi.status() == WL_CONNECTED) {

    HTTPClient http;
    http.begin(serverURL);

    int code = http.GET();

    if (code > 0) {
      Serial.print("HTTP Code: ");
      Serial.println(code);

      if (code == 200) {
        String payload = http.getString();
        Serial.println(payload);

        // SAME LOGIC (no change)
        if (payload.indexOf("Occupied") != -1) {
          Serial.println("LED ON");
          digitalWrite(LIGHT_PIN, HIGH); // LED ON
        } else {
          Serial.println("LED OFF");
          digitalWrite(LIGHT_PIN, LOW);  // LED OFF
        }
      }

    } else {
      Serial.print("HTTP Error: ");
      Serial.println(http.errorToString(code));

      // Safety OFF
      digitalWrite(LIGHT_PIN, LOW);
    }

    http.end();

  } else {
    Serial.println("WiFi Disconnected");

    // Safety OFF
    digitalWrite(LIGHT_PIN, LOW);
  }

  delay(5000); // 5 sec delay
}
